\missiontitle{Mission 1: Grind}

\teaser{The armies on campaign grapple with each other in brutal
  deadlock.}

%%----------------------------------------------
\begin{tablesetup}
  \dawnofwar

  \bigskip%
  Place one primary objective marker at table center, and two more
  each 12'' from a short table edge and 24'' from the long edges.
\end{tablesetup}

%%----------------------------------------------
\begin{missionrules}

\nightfighting

\missionsubheading{Continual Struggle.} Starting with Turn 2, at the
end of each player's turn they score 1 \emph{mission point} for each
primary objective marker they control.
\end{missionrules}


%%----------------------------------------------
\begin{scoring}
  
\begin{primaries}
  At game end compare the players' mission points and award victory
  points to the high and low scorer as follows:

  \bigskip%
  \centerline{%
    \begin{tabular}{lcc}
      {\bf Condition} &{\bf High} & {\bf Low} \\
      \hline
      Low scorer has no mission points & 9 & 0 \\
      Low scorer has at most half the mission points of the high scorer & 7 & 2 \\
      Low scorer has fewer mission points than the high scorer & 5 & 4 \\
      Both players have equal points & 4 & 4
    \end{tabular}}

\end{primaries}

\begin{secondaries}

\item \textit{Stalwart.} Score~2 victory points for each primary
  objective marker held at game end.

  \seizeground

  \seekanddestroy

  \meatgrinder
\end{secondaries}

\end{scoring}
